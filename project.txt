CREATE TABLE products (
    product_id INTEGER PRIMARY KEY,
    product_name VARCHAR(50) NOT NULL,
    categories VARCHAR(30),
    type_of_product VARCHAR(30),
    price INTEGER,
    shp_id FOREIGN KEY REFERENCES shipping_service(ship_id),
    inventory_id FOREIGN KEY REFERENCES inventory(inventory_id),
    store_id FOREIGN KEY REFERENCES mystore(store_id),
    sd_id FOREIGN KEY REFERENCES sales_data(sd_id),
    customer_id FOREIGN KEY REFERENCES customer(customer_id),
    rc_id FOREIGN KEY REFERENCES regular_customers(rc_id),
    manufacturer_id FOREIGN KEY REFERENCES manufacturer(man_id)
);

CREATE TABLE  sales_data (
    sd_id INTEGER PRIMARY KEY,
    region VARCHAR(30) NOT NULL,
    num_of_sales INTEGER,
    product_id FOREIGN KEY REFERENCES products(product_id)
);

CREATE TABLE contract(
    contract_id INTEGER PRIMARY KEY,
    card_number INTEGER,
    purchase_counter INTEGER,
    total_price INTEGER,
    store_id FOREIGN KEY REFERENCES mystore(store_id),
    rc_id FOREIGN KEY REFERENCES regular_customers(rc_id)

);

CREATE TABLE customer(
    customer_id INTEGER PRIMARY KEY,
    customer_f_name VARCHAR(30) NOT NULL,
    customer_l_name VARCHAR(30) NOT NULL,
    address VARCHAR(54),
    country VARCHAR(30),
    ship_id FOREIGN KEY REFERENCES shipping_service(ship_id),
    store_id FOREIGN KEY REFERENCES mystore(store_id),
    product_id FOREIGN KEY REFERENCES products(product_id)
);

CREATE TABLE inventory(
    inventory_id INTEGER PRIMARY KEY,
    num_of_allproducts INTEGER,
    sold_products VARCHAR,
    new_products VARCHAR,
    product_id FOREIGN KEY REFERENCES products(product_id)
);

CREATE TABLE mystore(
    store_id INTEGER PRIMARY KEY,
    product_id FOREIGN KEY REFERENCES products(product_id),
    employee_id FOREIGN KEY REFERENCES employee(emp_id),
    customer_id FOREIGN KEY REFERENCES customer(customer_id),
    contract_id FOREIGN KEY REFERENCES contract(contract_id),
    rc_id FOREIGN KEY REFERENCES regular_customers(rc_id)
);

CREATE TABLE employee(
    emp_id INTEGER PRIMARY KEY,
    emp_f_name VARCHAR(30) NOT NULL,
    emp_l_name VARCHAR(30) NOT NULL,
    specialization VARCHAR,
    phone VARCHAR (20),
    store_id FOREIGN KEY REFERENCES mystore(store_id)
);

CREATE TABLE shipping_service(
	ship_id INTEGER PRIMARY KEY,
	tracking_number INTEGER,
	ship_name VARCHAR(50),
	day_sending DATE,
	day_receiving DATE,
	product_id FOREIGN KEY REFERENCES products(product_id),
	customer_id FOREIGN KEY REFERENCES customer(customer_id),
	store_id FOREIGN KEY REFERENCES mystore(store_id),
	rc_id FOREIGN KEY REFERENCES regular_customers(rc_id),
);

CREATE TABLE regular_customer(
	rc_id INTEGER PRIMARY KEY,
	rc_f_name VARCHAR(30) NOT NULL,
	rc_l_name VARCHAR(30) NOT NULL,
	address VARCHAR(54),
    country VARCHAR(30),
    ship_id FOREIGN KEY REFERENCES shipping_service(ship_id),
    store_id FOREIGN KEY REFERENCES mystore(store_id),
    product_id FOREIGN KEY REFERENCES products(product_id),
	contract_id FOREIGN KEY REFERENCES contract(contract_id)
);

CREATE TABLE manufacturer(
	man_id INTEGER PRIMARY KEY,
	man_name VARCHAR(50),
)
